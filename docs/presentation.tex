% Slide 6: ACID Transactions
\begin{frame}{ACID Transactions}
    \textbf{Problem:} Money could "vanish" if service crashes mid-transfer
    
    \vspace{0.5cm}
    \textbf{Solution: Atomic Database Transactions}
    \begin{verbatim}
BEGIN TRANSACTION;
  UPDATE accounts SET balance = balance - 500 
    WHERE account_id = 'A';
  UPDATE accounts SET balance = balance + 500 
    WHERE account_id = 'B';
  INSERT INTO ledger_entries (...) VALUES (...);
COMMIT;
    \end{verbatim}
    
    \vspace{0.3cm}
    \textbf{Implementation:}
    \begin{itemize}
        \item PostgreSQL transactions ensure atomicity
        \item All steps succeed or none do
        \item Automatic rollback on failure
        \item Guarantees data consistency
    \end{itemize}
\end{frame}

% Slide 7: Idempotency
\begin{frame}{Idempotency}
    \textbf{Problem:} Duplicate requests due to client retries or network failures
    
    \vspace{0.5cm}
    \textbf{Solution: Idempotency Keys}
    \begin{verbatim}
Client Request:
{
  "idempotency_key": "unique-key-123",
  "amount": 500,
  "from_account": "A",
  "to_account": "B"
}

Service Logic:
1. Check if idempotency_key exists
2. If exists → return cached response
3. If new → process transaction and store result
    \end{verbatim}
    
    \textbf{Benefits:}
    \begin{itemize}
        \item Prevents duplicate transactions
        \item Safe retry mechanism
        \item Guarantees funds deducted exactly once
    \end{itemize}
\end{frame}

% Slide 8: Event-Driven Architecture
\begin{frame}{Event-Driven Architecture}
    \textbf{Decoupling Services:} Core Banking does not depend on Notification Service availability
    
    \vspace{0.5cm}
    \textbf{Flow:}
    \begin{enumerate}
        \item Transaction completes in Core Banking
        \item Event published to Kafka: \texttt{transaction-events}
        \item Notification Service consumes event asynchronously
        \item Sends Email/SMS + Generates PDF Receipt
    \end{enumerate}
    
    \vspace{0.5cm}
    \textbf{Benefits:}
    \begin{itemize}
        \item High availability
        \item Faster client responses
        \item Scalable notification processing
        \item Loose coupling between services
    \end{itemize}
\end{frame}

% Slide 9: Database Architecture
\begin{frame}{Database Architecture}
    \textbf{Read/Write Replication}
    \begin{itemize}
        \item \textbf{Primary Database}: Handles all write operations
        \item \textbf{Replica Database}: Handles read operations
    \end{itemize}
    
    \vspace{0.3cm}
    \textbf{Benefits:}
    \begin{itemize}
        \item Improved read performance
        \item High availability
        \item Load distribution
    \end{itemize}
    
    \vspace{0.5cm}
    \textbf{Data Models:}
    \begin{itemize}
        \item Users: Authentication and profile data
        \item Accounts: Financial accounts with balances
        \item Transactions: Transaction records with status
        \item Ledger Entries: Double-entry bookkeeping
        \item Payment Methods: User payment options
        \item Subscriptions: Recurring payment plans
    \end{itemize}
\end{frame}
